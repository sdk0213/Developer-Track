    if(N==1){
        System.out.println("1번째 디스크를 " + start + " -> " + to);
    }

    // 해당 변수는 목적지가 변경되었기 때문에 사용한다. (목적지가 가운데일수도 있고 맨 끝에일수도 있다.)

    HanoisTop(N-1, start, other);

    System.out.println(N+"번째 디스크를 " + start + " -> " + to);

    HanoisTop(N-1, other, to);
